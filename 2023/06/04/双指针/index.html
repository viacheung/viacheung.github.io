class Solution {
    public int removeElement(int[] nums, int val) {
        int left=0,right=nums.length-1;
        while(left<=right){
            if(nums[left]==val){
                nums[left]=nums[right];
                right--;
            }else{
                left++;
            }
        }
        return left;
    }

class Solution {
    public int[] sortedSquares(int[] nums) {
        int n = nums.length;
        int[] ans = new int[n];
        //æ³¨æ„i <= j;åé¢æœ‰åˆ†å·
        for (int i = 0, j = n - 1, pos = n - 1; i <= j;) {
            if (nums[i] * nums[i] > nums[j] * nums[j]) {
                ans[pos] = nums[i] * nums[i];
                ++i;
            } else {
                ans[pos] = nums[j] * nums[j];
                --j;
            }
            --pos;
        }
        return ans;
    }
}

class Solution {

        int n = nums.length;
        //è¿™ä¸ªåˆ¤æ–­æ¡ä»¶ä¸å†™ä¹Ÿå¯ä»¥é€šè¿‡
        if (n == 0) {
            return 0;

    }
